exit

exit

exit
vlan 205
name Unused-Ports
exit
vlan 201
name New-Native
exit

! Configure Interfaces
interface fastEthernet 0/1
switchport mode access
switchport access vlan 10
exit
interface fastEthernet 0/3
switchport mode access
switchport access vlan 20
exit
interface fastEthernet 0/2
switchport mode access
switchport access vlan 30
exit
interface gigabitEthernet 0/1
switchport mode trunk
switchport nonegotiate
switchport trunk allowed vlan 10,20,30,90,100
switchport trunk native vlan 201
exit

interface vlan 100
ip address 10.11.12.74 255.255.255.248
no shutdown
exit

! Adding Default Gateway
ip default-gateway 10.11.12.73

! Adding All Unused Ports in Different VLAN
interface range fastEthernet 0/4-24, gigabitEthernet 0/2
switchport access vlan 205
shutdown
exit

! Enabling Port Security
interface range fastEthernet 0/1-3
switchport port-security
switchport port-security maximum 1
switchport port-security mac-address sticky
switchport port-security violation restrict
exit

ip dhcp snooping vlan 10,20,30,90
ip arp inspection vlan 10,20,30,90
ip arp inspection validate src-mac
ip arp inspection validate dst-mac
ip arp inspection validate ip

interface gigabitEthernet 0/1
ip dhcp snooping trust
ip arp inspection trust
exit

interface range fastEthernet 0/1-3
ip dhcp snooping limit rate 6
spanning-tree portfast
spanning-tree bpduguard enable
exit

! 802.1X Authentication
aaa new-model
radius-server host 10.11.12.34 key cisco

aaa authentication dot1x default group radius

dot1x system-auth-control

interface range fastEthernet 0/1-3
authentication port-control auto
dot1x pae authenticator
exit

! Enabling Timestamps in Logs
service timestamps log datetime msec

! Sending Logs to SysLog Server
logging on
logging 10.11.12.51

! Enabling NTP Server
ntp server 10.11.12.50
ntp authenticate
ntp trusted-key 1
ntp authentication-key 1 md5 cisco

! Save Configration
do write

# S19 #
#######
! Basic Configration
enable
configure terminal

hostname S19

banner motd &#### Authorized Access Only ####&

username cisco privilege 15 secret cisco

ip domain-name cisco.local
crypto key generate rsa general-keys modulus 1024

line vty 0 4
login local
transport input ssh
exit

line console 0
login local
exit

enable secret cisco

service password-encryption

! Creating VLANs
vlan 500
name Unused-Ports
exit

! Configure Interfaces
interface range fastEthernet 0/1-2
switchport mode access
exit

interface vlan 1
ip address 10.11.12.186 255.255.255.248
no shutdown
exit

! Adding Default Gateway
ip default-gateway 10.11.12.185

! Adding All Unused Ports in Different VLAN
interface range fastEthernet 0/3-24, gigabitEthernet 0/2
switchport access vlan 500
shutdown
exit

! Enabling Port Security
interface range fastEthernet 0/1-2
switchport port-security
switchport port-security maximum 1
switchport port-security mac-address sticky
switchport port-security violation restrict
exit

ip dhcp snooping
ip arp inspection vlan 1
ip arp inspection validate src-mac
ip arp inspection validate dst-mac
ip arp inspection validate ip

interface gigabitEthernet 0/1
ip dhcp snooping trust
ip arp inspection trust
exit

interface range fastEthernet 0/1-2
ip dhcp snooping limit rate 6
spanning-tree portfast
spanning-tree bpduguard enable
exit

! Enabling Timestamps in Logs
service timestamps log datetime msec

! Save Configration
do write
